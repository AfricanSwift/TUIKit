    ,custom(bar: TUIBar):

    let c = [Character("0")]
let i = Character(".")
let cBar = TUIBar(format: [.percent, .text(" {"),
                           .complete(c), .incomplete(i),
                           .text("} "), .elapsed,
                           .text(" "), .message], 
                  width: 12)

var bar = TUIProgress(style: .custom(bar: cBar), message: "Hello world!")
